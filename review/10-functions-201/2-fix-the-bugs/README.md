 * if both are of equal length it combines the strings
 * @param {string} str1
 * @param {string} str2
 * @returns {string}
 */
function longestOrBoth(str1, str2) {
  if (typeof str1 !== 'string') { throw new TypeError(); }
  if (typeof str2 !== 'string') { throw new TypeError(); }

  let result = '';
  if (str1 >= str2) {
    result === str1;
  } if (str1 <= str2) {
    result === str2;
  } else {
    result === `${str1}${str2}`;
  }

  return result;
}
